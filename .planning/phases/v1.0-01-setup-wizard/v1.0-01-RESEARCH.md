## Summary

The Setup Wizard is a Node.js CLI script (`scripts/setup.js`) invoked via `npm run setup`. It guides users through collecting all required `.env` values using interactive Inquirer.js prompts, validates every field with Zod before touching the file system, shows a confirmation summary, then atomically writes `.env`.

The project already uses `dotenv@17.4+` and Node.js. The wizard adds `@inquirer/prompts` (the modern scoped package from Inquirer.js 13.x), `zod@4.3+` for schema validation, and `chalk@4.x` (CommonJS-compatible, since the project currently targets CommonJS via `require`).

The critical constraint is that the project has **no build step for Node.js scripts** — `scripts/build-env.js` uses `require()` throughout, confirming CommonJS. This rules out `chalk@5.x` (ESM-only) and requires verifying that Inquirer's scoped packages work as CommonJS.

**Primary recommendation:** Write `scripts/setup.js` as a self-contained CommonJS script using `@inquirer/prompts` (supports CJS via dynamic import shim), `zod` (CJS-compatible), and `chalk@4.x`. Keep the wizard in `scripts/` alongside `build-env.js` — no new top-level `bin/` or `lib/` directories needed for this phase.

### Pattern 1: Collect-Validate-Confirm-Write

**What:** Gather all inputs via prompts → validate entire object with Zod → show summary → confirm → write atomically. Never write partial state.

**When to use:** Every setup or update operation.

**Example:**
```javascript
// scripts/setup.js (CommonJS)
const { input, confirm, select } = require('@inquirer/prompts');
const { z } = require('zod');
const chalk = require('chalk');
const envParser = require('./lib/env-parser');

const ConfigSchema = z.object({
  FORWARD_TO_EMAIL: z.string().email('Must be a valid email address'),
  ALLOWED_SENDERS: z.string().optional(),
  ALLOWED_FILE_TYPES: z.string().default('pdf'),
  DRY_RUN: z.literal('true').default('true'),
  ENABLE_LIVE_FORWARDING: z.literal('false').default('false'),
});

async function main() {
  // Load existing .env if present
  const existing = envParser.load();

  // Collect inputs
  const forwardTo = await input({
    message: 'Forward-to email address:',
    default: existing.FORWARD_TO_EMAIL,
    validate: (v) => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(v) || 'Invalid email format',
  });

  // ... more prompts ...

  // Validate entire config
  const config = ConfigSchema.parse({ FORWARD_TO_EMAIL: forwardTo, ... });

  // Summary + confirm
  console.log(chalk.bold('\nConfiguration summary:'));
  console.log(chalk.cyan(`  Forward to: ${config.FORWARD_TO_EMAIL}`));
  const ok = await confirm({ message: 'Write .env now?' });
  if (!ok) { console.log('Aborted — nothing written.'); return; }

  // Atomic write
  envParser.write(config);
  console.log(chalk.green('✓ .env written successfully.'));
  console.log('Next: npm run push — then run bootstrapProperties in Apps Script editor.');
}
main().catch(console.error);
```

### Pattern 2: Atomic `.env` Write (Preserve Comments)

**What:** Parse existing `.env` line-by-line, update only changed keys, preserve all comments and unknown keys, write to a `.env.tmp` first, then rename.

**When to use:** Every write operation — first-time setup and updates.

**Example:**
```javascript
// scripts/lib/env-parser.js
const fs = require('fs');
const path = require('path');

const ENV_PATH = path.resolve(__dirname, '../../.env');

function load() {
  if (!fs.existsSync(ENV_PATH)) return {};
  const content = fs.readFileSync(ENV_PATH, 'utf8');
  const result = {};
  for (const line of content.split('\n')) {
    const match = line.match(/^([A-Z_]+)\s*=\s*(.*)$/);
    if (match) result[match[1]] = match[2].trim();
  }
  return result;
}

function write(updates) {
  let lines = [];
  if (fs.existsSync(ENV_PATH)) {
    lines = fs.readFileSync(ENV_PATH, 'utf8').split('\n');
  }

  const written = new Set();
  const patched = lines.map((line) => {
    const match = line.match(/^([A-Z_]+)\s*=/);
    if (match && updates[match[1]] !== undefined) {
      written.add(match[1]);
      return `${match[1]}=${updates[match[1]]}`;
    }
    return line;
  });

  // Append any new keys not already in file
  for (const [key, val] of Object.entries(updates)) {
    if (!written.has(key)) patched.push(`${key}=${val}`);
  }

  const tmp = ENV_PATH + '.tmp';
  fs.writeFileSync(tmp, patched.join('\n'), 'utf8');
  fs.renameSync(tmp, ENV_PATH);  // Atomic on same filesystem
}

module.exports = { load, write };
```

### Pattern 3: Keyword Generation from Use Case + Languages

**What:** Given a free-text use-case description and selected language list, emit appropriate `SUBJECT_KEYWORDS` and `ATTACHMENT_FILENAME_KEYWORDS` strings.

**When to use:** SETUP-03 — after user describes their use case.

**Implementation approach:** A simple lookup table mapping language codes to invoice/receipt keyword sets (as used in `src/constants.js`). User selects languages, wizard combines the keyword sets.

```javascript
// scripts/lib/keyword-gen.js
const KEYWORDS_BY_LANG = {
  en: { subject: ['invoice', 'receipt', 'bill', 'statement'], attachment: ['invoice', 'receipt'] },
  fr: { subject: ['facture', 'reçu', 'relevé'], attachment: ['facture'] },
  es: { subject: ['factura', 'recibo', 'comprobante'], attachment: ['factura'] },
  pt: { subject: ['fatura', 'recibo', 'comprovante'], attachment: ['fatura'] },
  de: { subject: ['rechnung', 'quittung', 'beleg'], attachment: ['rechnung'] },
  zh: { subject: ['发票', '收据', '账单'], attachment: ['发票'] },
  ja: { subject: ['請求書', '領収書', '明細'], attachment: ['請求書'] },
};

function generate(languages) {
  const subject = new Set(), attachment = new Set();
  for (const lang of languages) {
    const kw = KEYWORDS_BY_LANG[lang] || {};
    (kw.subject || []).forEach(k => subject.add(k));
    (kw.attachment || []).forEach(k => attachment.add(k));
  }
  return {
    SUBJECT_KEYWORDS: [...subject].join(','),
    ATTACHMENT_FILENAME_KEYWORDS: [...attachment].join(','),
  };
}

module.exports = { generate };
```

### Anti-Patterns to Avoid

- **Writing `.env` before validation:** Always validate full config with Zod before any file I/O.
- **Non-atomic writes:** Never truncate `.env` and re-write — use tmp + rename.
- **Overwriting comments:** Parse line-by-line and patch only known keys.
- **chalk@5.x in a CommonJS project:** This project uses `require()` — stick with chalk@4.x.
- **Calling Apps Script directly from CLI:** Not possible without complex OAuth. CLI only writes `.env` and instructs user to run `npm run push`.

## Don't Hand-Roll

| Problem | Don't Build | Use Instead | Why |
|---------|-------------|-------------|-----|
| Interactive prompts | Custom readline loop | `@inquirer/prompts` | Handles TTY, SIGINT, validation, defaults, masking |
| Email validation | Custom regex | `zod.string().email()` | RFC-compliant, handles edge cases |
| Config schema validation | Manual key checks | Zod | Type inference, clear error messages, composable |
| `.env` parsing | Manual split/parse | Existing `dotenv` + custom patch writer | dotenv handles edge cases; patch writer handles comments |

**Key insight:** The `.env` format has many edge cases (quoted values, `#` in values, trailing spaces, Windows line endings). Use dotenv for reading, but write with a careful line-by-line patcher to preserve structure.

## Common Pitfalls

### Pitfall 1: ESM/CommonJS mismatch with Inquirer 13.x and chalk 5.x

**What goes wrong:** `@inquirer/prompts` 13.x requires ESM in some configurations; chalk 5.x is ESM-only. The project uses `require()` (CommonJS). Mixing these gives "ERR_REQUIRE_ESM".

**Why it happens:** Inquirer rewrote to ESM first in v9+; chalk went ESM-only in v5.

**How to avoid:** Use chalk@4.x (last CommonJS version). Verify `@inquirer/prompts` works with `require()` — in practice, recent versions provide a CJS build, but test with `node -e "require('@inquirer/prompts')"` immediately after install.

**Warning signs:** `ERR_REQUIRE_ESM` on `require('chalk')` or `require('@inquirer/prompts')`.

### Pitfall 2: Silent dotenv failures (missing keys → undefined)

**What goes wrong:** dotenv loads `.env` but missing keys return `undefined` silently. Apps Script crashes later with cryptic errors.

**How to avoid:** Validate with Zod after every `.env` write. Run `ConfigSchema.parse(dotenv.config().parsed)` to catch missing/malformed keys.

### Pitfall 3: Non-atomic `.env` writes corrupt the file

**What goes wrong:** Script crashes mid-write. `.env` is half-written — broken state.

**How to avoid:** Write to `.env.tmp`, validate the result, then `fs.renameSync` (atomic on same filesystem). If rename fails, `.env` is untouched.

### Pitfall 4: Re-run destroys `.env` comments

**What goes wrong:** Wizard reads all values, re-writes entire file from scratch → all user comments and blank lines lost.

**How to avoid:** Parse `.env` line-by-line, update only keys present in the update object, preserve everything else.

### Pitfall 5: `ALLOWED_SENDERS` format mismatch with Apps Script

**What goes wrong:** Wizard writes `ALLOWED_SENDERS=sender@example.com, other@example.com` (with spaces after commas). Apps Script's `_getList()` in `config.js` splits on `,` and trims, so it works — but the wizard should be consistent.

**How to avoid:** Normalize comma-separated fields: trim each value, join with `,` (no spaces). Validate each item as an email before joining.

## Open Questions

1. **`@inquirer/prompts` CommonJS compatibility**
   - What we know: Inquirer 13.x is ESM-first; the scoped `@inquirer/prompts` package ships both CJS and ESM exports in recent versions.
   - What's unclear: Whether `require('@inquirer/prompts')` works without a build step in this project's Node version.
   - Recommendation: First task is to `npm install @inquirer/prompts` and run `node -e "const i = require('@inquirer/prompts'); console.log(typeof i.input)"` to confirm before writing wizard code.

2. **Keyword vocabulary for non-invoice use cases**
   - What we know: `src/constants.js` has a 7-language invoice vocabulary. SETUP-03 asks the wizard to generate keywords based on user's use case.
   - What's unclear: How to handle use cases outside "supplier invoices" (e.g., payslips, shipping notifications).
   - Recommendation: For Phase v1.0-01, scope to a curated keyword map per language for the common use cases (invoices/receipts as default). Let user edit `.env` directly for exotic use cases. Document this limitation.
