<feature name="env-parser">
  <files>scripts/lib/env-parser.js, scripts/lib/__tests__/env-parser.test.js</files>
  <behavior>
`load()`: Reads `{projectRoot}/.env`. Returns `{}` if file does not exist. Parses `KEY=VALUE` lines; trims whitespace from values. Ignores blank lines and lines starting with `#`.

`write(updates)`: Takes an object `{ KEY: 'value', ... }`. Reads existing `.env` line-by-line. For each line matching `KEY=`, replaces the value if `updates[KEY]` is defined. Appends any keys in `updates` not already in the file. Writes to `.env.tmp` first, then calls `fs.renameSync('.env.tmp', '.env')`. If file does not exist, creates it from scratch.

Edge cases:
- Lines with `KEY=` (empty value) should be updated correctly
- Keys present in existing file but NOT in `updates` must be preserved unchanged
- Comment lines (`# ...`) and blank lines must survive a `write()` call unchanged
- If `fs.renameSync` throws, the original `.env` must be untouched
  </behavior>
  <implementation>
Use Node.js built-ins only: `fs`, `path`. CommonJS (`module.exports`). Resolve `.env` path relative to `process.cwd()` so it works from project root when invoked via `npm run setup`.

```js
// scripts/lib/env-parser.js
const fs = require('fs');
const path = require('path');

const ENV_PATH = path.resolve(process.cwd(), '.env');

function load() {
  if (!fs.existsSync(ENV_PATH)) return {};
  const result = {};
  for (const line of fs.readFileSync(ENV_PATH, 'utf8').split('\n')) {
    const m = line.match(/^([A-Z_][A-Z0-9_]*)\s*=\s*(.*)\s*$/);
    if (m) result[m[1]] = m[2].trim();
  }
  return result;
}

function write(updates) {
  let lines = fs.existsSync(ENV_PATH)
    ? fs.readFileSync(ENV_PATH, 'utf8').split('\n')
    : [];
  const written = new Set();
  const patched = lines.map(line => {
    const m = line.match(/^([A-Z_][A-Z0-9_]*)\s*=/);
    if (m && updates[m[1]] !== undefined) {
      written.add(m[1]);
      return `${m[1]}=${updates[m[1]]}`;
    }
    return line;
  });
  for (const [k, v] of Object.entries(updates)) {
    if (!written.has(k)) patched.push(`${k}=${v}`);
  }
  const tmp = ENV_PATH + '.tmp';
  fs.writeFileSync(tmp, patched.join('\n'), 'utf8');
  fs.renameSync(tmp, ENV_PATH);
}

module.exports = { load, write };
```

In tests, override `ENV_PATH` by writing to a temp directory path. Use `jest.spyOn(fs, 'renameSync')` to test atomic write behavior.
  </implementation>
</feature>

<feature name="validators">
  <files>scripts/lib/validators.js, scripts/lib/__tests__/validators.test.js</files>
  <behavior>
`isValidEmail(str)`: Returns `true` if `str` matches a basic email pattern (contains `@`, non-empty local/domain parts, has a dot in domain). Returns `false` otherwise. Must match what `src/config.js` expects (simple email strings, not RFC-strict).

`isValidEmailList(str)`: Splits `str` on commas, trims each part, calls `isValidEmail` on each non-empty part. Returns `true` if all pass. Returns `true` for an empty string (field is optional). Returns `false` if any non-empty part fails.

`isValidFileTypeList(str)`: Splits on commas, trims each part. Returns `false` for empty string. Returns `false` if any part contains a dot (e.g., `.pdf`). Returns `true` if all parts are non-empty alphanumeric strings.
  </behavior>
  <implementation>
```js
// scripts/lib/validators.js
function isValidEmail(str) {
  return typeof str === 'string' && /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(str.trim());
}

function isValidEmailList(str) {
  if (!str || !str.trim()) return true; // optional
  return str.split(',').map(s => s.trim()).filter(Boolean).every(isValidEmail);
}

function isValidFileTypeList(str) {
  if (!str || !str.trim()) return false;
  return str.split(',').map(s => s.trim()).filter(Boolean)
    .every(ext => ext.length > 0 && !ext.includes('.') && /^[a-zA-Z0-9]+$/.test(ext));
}

module.exports = { isValidEmail, isValidEmailList, isValidFileTypeList };
```

```js
// scripts/lib/keyword-gen.js
// Vocabulary aligned with src/constants.js DEFAULT_SUBJECT_KEYWORDS
const KEYWORDS_BY_LANG = {
  en: {
    subject: ['invoice', 'receipt', 'bill', 'statement', 'order confirmation'],
    attachment: ['invoice', 'receipt']
  },
  fr: {
    subject: ['facture', 'reçu', 'relevé', 'bon de commande'],
    attachment: ['facture', 'reçu']
  },
  es: {
    subject: ['factura', 'recibo', 'comprobante', 'pedido'],
    attachment: ['factura', 'recibo']
  },
  pt: {
    subject: ['fatura', 'recibo', 'comprovante', 'nota fiscal'],
    attachment: ['fatura', 'recibo']
  },
  de: {
    subject: ['rechnung', 'quittung', 'beleg', 'bestellung'],
    attachment: ['rechnung', 'quittung']
  },
  zh: {
    subject: ['发票', '收据', '账单', '订单确认'],
    attachment: ['发票', '收据']
  },
  ja: {
    subject: ['請求書', '領収書', '明細', '注文確認'],
    attachment: ['請求書', '領収書']
  },
};

function generate(languages) {
  const subject = new Set();
  const attachment = new Set();
  for (const lang of (languages || [])) {
    const entry = KEYWORDS_BY_LANG[lang];
    if (!entry) continue;
    entry.subject.forEach(k => subject.add(k));
    entry.attachment.forEach(k => attachment.add(k));
  }
  return {
    SUBJECT_KEYWORDS: [...subject].join(','),
    ATTACHMENT_FILENAME_KEYWORDS: [...attachment].join(','),
  };
}

module.exports = { generate, KEYWORDS_BY_LANG };
```
  </implementation>
